function calcularSumatoria ( miu,lamda, servidores){

    var i = 0;
    var sumatoria = 0;

    while (i <= servidores) {
        var numerador = 0;
        var denominador = 0;

        numerador = Math.pow( (lamda/miu) , i);
        denominador = calcularFactorial (servidores - 1); 

        sumatoria = sumatoria + (numerador/denominador);
        i++; 
    }
    return numerador; 
}

function calcularFactorial (servidores) {
    if (servidores == 0 ) {
        return 1;
    } else {
        return servidores * calcularFactorial (servidores - 1);
    }
}


function calcularPotencia (base, potencia) {
    var resultado = 1;
    for (var i=1;i<=potencia; i++ ) {
        resultado = resultado * base;
    }
    return resultado;

}
